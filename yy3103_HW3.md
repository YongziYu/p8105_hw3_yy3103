``` r
#load data
instacart =
  instacart %>%
  janitor::clean_names()
  
aisle=
  instacart %>%
  count(aisle) %>%
  arrange(desc(n))

aisle_plot=
  instacart %>%
  group_by(aisle) %>%
  summarise(
    count_order=n()
  ) %>%
  filter(count_order>10000)

ggplot(data=aisle_plot,aes(y=aisle,x=count_order))+geom_point()
```
